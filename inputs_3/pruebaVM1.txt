program test_vm_division_cero;
var
    resultado : int;
    x : int;
    y : int;

void dividir(a: int, b: int) [
    var temp : int;
    {
        print("Intentando dividir ", a, " entre ", b);
        temp = a / b;
        print("Resultado: ", temp);
        resultado = temp;
    }
];

main {
    x = 100;

end
