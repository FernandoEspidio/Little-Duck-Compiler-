program errores_lexicos_sintacticos;

# Prueba: errores lexicos controlados (simbolos invalidos, string sin cerrar)
# Prueba: errores sintacticos (expresion incompleta, parentesis sin cerrar, punto y coma faltante)

var
  x, y : int;
  r : float;
  datos : int[4];

int sumar(a: int, b: int) [
  {
    return a + b;
  }
];

main
{
  x = 10;
  y = 5;

  # Error lexico: simbolo ^ no pertenece al lenguaje
  x = 3 ^ 2;

  # Error lexico: simbolo ~ no pertenece al lenguaje
  y = 4 ~ 1;

  # Error lexico: simbolo @ no pertenece al lenguaje
  r = @;

  # Error lexico: string sin cerrar
  print("este string no cierra);

  # Error sintactico: expresion incompleta en asignacion
  x = ;

  # Error sintactico: falta cerrar parentesis en while
  while (x < 10
  {
    x++;
  };

  # Error sintactico: falta primer ; dentro del for
  for (x = 0 x < 4; x++)
  {
    print(datos[x]);
  };

  # Error sintactico: falta : despues de case
  switch (y % 2)
  {
    case 0
      print("par");
    case 1:
      print("impar");
  };

  # Error sintactico: falta ; al final del print
  print("resultado:", sumar(x, y))

  print("fin");
}
end
