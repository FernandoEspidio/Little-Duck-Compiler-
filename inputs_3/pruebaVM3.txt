program test_vm_indice_fuera_rango_escritura;
var
    arreglo : int[5];
    i : int;

main {
    print("=== Test VM: Índice Fuera de Rango (Escritura) ===");
    
    i = 0;
    while (i < 5) {
        arreglo[i] = i * 10;
        print("arreglo[", i, "] = ", arreglo[i]);
        i = i + 1;
    };
    
    print("Intentando escribir en índice inválido...");
    arreglo[10] = 999;
    print("Esta línea no debería ejecutarse");
}
end
