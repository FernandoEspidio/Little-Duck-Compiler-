program errores_estructurales_nuevos;

var
  x, y, z, i : int;

int duplicar(a: int) [
  {
    # Error sintactico: falta ';' despues del return
    return a * 2

    # Esta linea ayuda a observar la recuperacion despues del return mal cerrado
    print("despues del return sin punto y coma");
  }
];

void imprimirSeparador() [
  {
    print("----------------");
  }
];

main
{
  # Error sintactico: falta ';' despues de la asignacion
  x = 10
  y = 20;

  # Error sintactico: faltaria un operador entre x y y
  z = x y;

  # Error sintactico: condicion vacia dentro del if
  if ()
  {
    print("condicion vacia");
  };

  # Error sintactico: else sin un if valido asociado
  else
  {
    print("else suelto");
  };

  # Error sintactico: llamada a funcion sin punto y coma final
  imprimirSeparador()

  # Error sintactico: el for no tiene condicion entre los dos ';'
  # En esta gramatica for_init y for_update pueden ser vacios,
  # pero la expresion central es obligatoria.
  for (i = 0; ; i++)
  {
    print(i);
  };

  # Error sintactico: estructura do-while incompleta
  # Despues del cuerpo deberia venir keyword_while.
  do
  {
    x++;
  }
  (x < 15);

  # Error sintactico: default aparece antes de los case
  # Segun la gramatica, switch requiere primero CASE_LIST y despues DEFAULT_OPT.
  switch (x)
  {
    default:
      print("default primero");
    case 1:
      print("uno");
  };

  # Error sintactico: break sin punto y coma
  while (x < 20)
  {
    break
    x++;
  };

  # Error sintactico: falta coma entre argumentos del print
  print("valores", x y);

  # Esta linea final sirve para revisar si el parser logro recuperarse
  print("fin del cuarto caso de errores");
}
end
